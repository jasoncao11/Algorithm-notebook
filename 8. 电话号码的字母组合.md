**给定一个仅包含数字 2-9 的字符串，返回所有它能表示的字母组合。答案可以按 任意顺序 返回。给出数字到字母的映射如下（与电话按键相同）。注意 1 不对应任何字母。**

```
class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        #回溯法
        if not digits:
            return list()
        
        phoneMap = {
            "2": "abc",
            "3": "def",
            "4": "ghi",
            "5": "jkl",
            "6": "mno",
            "7": "pqrs",
            "8": "tuv",
            "9": "wxyz",
        }
        res= []
        n = len(digits)
        #将要遍历的字符串索引
        ind=0
        #当前字母组合
        com = []
        #构造递归
        def dfs(ind, com):
            #递归终止条件
            if ind == n:
                res.append(''.join(com))
                return

            for c in phoneMap[digits[ind]]:
                dfs(ind+1, com+[c])
        #调用递归
        dfs(ind, com)
        #返回要优化的目标
        return res
```
